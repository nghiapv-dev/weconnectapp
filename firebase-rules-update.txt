# Firebase Firestore Security Rules
# Copy these rules to your Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - allow authenticated users to read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow read: if request.auth != null; // Allow reading other users for chat
    }
    
    // Chats collection - allow authenticated users to read/write
    match /chats/{chatId} {
      allow read, write: if request.auth != null;
    }
    
    // UserChats collection - allow authenticated users to read/write their own data
    match /userChats/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}

# Steps to apply:
# 1. Go to Firebase Console: https://console.firebase.google.com
# 2. Select your project
# 3. Go to Firestore Database > Rules
# 4. Replace the existing rules with the rules above
# 5. Click "Publish"
